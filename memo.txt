Complex.h
#pragma once
#include <cstdio>
class Complex
{
	double real;
	double imag;
public:
	void setComplex(double r, double i) {
		real = r;
		imag = i;
	}
	void readComplex(const char* msg = "복소수= ");
	void printComplex(const char* msg = "복소수= ");
	void addComplex(Complex a, Complex b);
};


Complex.cpp
#include "Complex.h"
void Complex::readComplex(const char* msg) {
	printf("%s", msg);
	scanf_s("%lf%lf", &real, &imag);
}
void Complex::printComplex(const char* msg) {
	printf("%s %4.2f + %4.2fi\n", msg, real, imag);
}
void Complex::addComplex(Complex a, Complex b) {
	real = a.real + b.real;
	imag = a.imag + b.imag;
}
